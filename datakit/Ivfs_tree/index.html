Module Ivfs_tree

Expose Irmin internal trees (to be upstreamed).

Irmin's public API exposes a key/value store, but we need to see directory nodes too. This module uses Irmin's private API to get that information.

module Path : Irmin.Path.S with type Path.step = string
type step = Path.step
type path = Path.t
type perm = [
| `Normal
| `Exec
| `Link
]
module type S : Irmin.S with type S.key = path and type S.contents = Ivfs_blob.t and type S.branch = string and type S.step = step and type S.metadata = perm
module type MAKER : functor (C : Irmin.Contents.S) -> functor (P : Irmin.Path.S) -> functor (B : Irmin.Branch.S) -> Irmin.S with type MAKER.key = P.t and type MAKER.step = P.step and module MAKER.Key = P and type MAKER.contents = C.t and type MAKER.branch = B.t and type MAKER.metadata = perm
module Make : functor (M : MAKER) -> S
