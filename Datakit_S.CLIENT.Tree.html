Module Datakit_S.CLIENT.Tree

module Tree: Datakit_S.READABLE_TREE  with type 'a or_error := 'a or_error
A read-only tree of files, directories and symlinks.
type t 
type +'a or_error 
val read : t ->
       Datakit_path.t ->
       [ `Dir of string list | `File of Cstruct.t | `Link of string ]
       or_error Lwt.t
read t path is the contents of the object at the path.
val stat : t ->
       Datakit_path.t ->
       Datakit_S.stat option or_error Lwt.t
stat t path is the metadata of the object at path.
val exists : t ->
       Datakit_path.t -> bool or_error Lwt.t
exists t path is true if stat t path isn't None.
val exists_file : t ->
       Datakit_path.t -> bool or_error Lwt.t
exists_file t path is similar to Datakit_S.READABLE_TREE.exists but for files only.
val exists_dir : t ->
       Datakit_path.t -> bool or_error Lwt.t
exists_dir t path is similar to Datakit_S.READABLE_TREE.exists but for directories only.
val read_file : t ->
       Datakit_path.t -> Cstruct.t or_error Lwt.t
read_file t path resolves path to a file, or returns an error if it isn't a file.
val read_dir : t ->
       Datakit_path.t -> string list or_error Lwt.t
read_dir t path resolves path to a directory, or returns an error if it isn't one.
val read_link : t ->
       Datakit_path.t -> string or_error Lwt.t
read_link t path resolves path to a symlink, or returns an error if it isn't one.
